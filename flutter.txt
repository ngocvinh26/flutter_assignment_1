import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'App Manager',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: const LoginScreen(),
    );
  }
}

// Trang đăng nhập
class LoginScreen extends StatefulWidget {
  const LoginScreen({super.key});

  @override
  _LoginScreenState createState() => _LoginScreenState();
}

class _LoginScreenState extends State<LoginScreen> {
  bool isLoggedIn = false;

  void loginAsManager() {
    setState(() {
      isLoggedIn = true;
    });
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => const ManagerScreen(),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Login Screen'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            const Text('Chưa đăng nhập'),
            const SizedBox(height: 20),
            ElevatedButton(
              onPressed: loginAsManager,
              child: const Text('Đăng nhập với vai trò Manager'),
            ),
          ],
        ),
      ),
    );
  }
}

// Trang Manager (đã đăng nhập)
class ManagerScreen extends StatefulWidget {
  const ManagerScreen({super.key});

  @override
  _ManagerScreenState createState() => _ManagerScreenState();
}

class _ManagerScreenState extends State<ManagerScreen> {
  bool showTeamInfo = true;

  @override
  void initState() {
    super.initState();
    // Hiển thị thông tin nhóm trong 10 giây
    Future.delayed(const Duration(seconds: 10), () {
      setState(() {
        showTeamInfo = false;
      });
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Manager Screen'),
      ),
      body: Center(
        child: showTeamInfo
            ? const Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Text('Thông tin nhóm:'),
                  Text('Thành viên 1: Nguyễn Ngọc Vinh'),
                  
                ],
              )
            : const Text('Đã đăng nhập với vai trò Manager'),
      ),
    );
  }
}
